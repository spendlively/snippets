
//Обрезать по времени 10с (-t 10)
ffmpeg -i input.mp4 -t 10 output.mp4

//Обрезать по времени 10с начиная с 15с
(-ss 15 указать до входного файла!)
ffmpeg -ss 15 -i input.mp4 -t 10 output.mp4

//Обрезать по времени 10.5с начиная с 00:17:34 (или 00:17:34.5)
(-ss 15 указать до входного файла!)
ffmpeg -ss 00:17:34 -i input.mp4 -t 10.5 output.mp4

//Обрезать последние 10с видео
(-sseof 10 указать до входного файла!)
ffmpeg -sseof 10 -i input.mp4 output.mp4
