
# область видимости может быть встроенной, глобальной и локальной

# встроенное просранство имен  - built in

# правила поиска имени
# 1. локальная область (local)
# 2. объемлющая область (enclosing) (локальная в кот. вложена другая локальная)
# 3. глобальная (global)
# 4. встроенная (built in) (область встроенных функций)

# имя из встроенной области можно переопределить

# явное указание области
# глобальная
# global a

# локальная на уровень выше
# nonlocal а


g = ''
x = 'global'
def foo():
    x = 'enclosing'
    def bar():

        global g
        #global x
        #nonlocal x
        x = 'local'

        g += x

    bar()

foo()

print(g)
