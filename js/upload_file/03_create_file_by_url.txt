
function getFileByUrl(url, filename) {

    let xhr = new XMLHttpRequest();
    xhr.open('GET', url, true);
    xhr.responseType = 'blob';
    xhr.onload = function (e) {
        if (this.status === 200) {
            let blob = this.response,
                file = new File([blob], filename)
            console.log(file)
        }
    };
    xhr.send();
}
