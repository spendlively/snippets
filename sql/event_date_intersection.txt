//1. Найти ивенты которые пересекаются с моим ивентом по времени

//Ищу ивент который
 - начался раньше конца моего ивента и
 - закончился позже начала моего ивента

select count(*) from tbl
where tbl.start < :end && tbl.end > :start



//2. Найти все ивенты которые не пересекаются с моим ивентом
 - закончились до начала моего ивента
 - начались после конца моего ивента

select count(*) from tbl
where tbl.end < :start or tbl.start > :end
