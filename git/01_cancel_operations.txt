
//1. Отмена непроиндексированных изменений (до последнего коммита)
//CHECKOUT
git checkout -- file.txt


//2. Отмена добавления файла в индекс
//RESET
git reset HEAD file.txt


//3. Правка уже сделанного коммита (добавление изменений в предыдущий коммит)
//--AMEND
git add file.txt
git commit --amend
//можно просто изменить описание коммита, если не добавлять никаких изменений


//4. Отмена незапушенного коммита
//RESET
//с сохранением изменений (до состояния проиндексированных изменений перед коммитом)
git reset --soft HEAD^
//с удалением изменений (до состояния предыдущего коммита, с удалением изменений)
git reset --hard HEAD^


//5. Отмена запушенного коммита
//REVERT
git revert a11e1a6t3f670d734ef9d9e7428d24038d668b4t
git push


//6. Отмена незапушенного мерджа
git reset --hard HEAD^
git pull
(Нельзя выполнить сброс в режиме «мягкий» во время слияния)


//7. Переключение между коммитами
//Переключиться на нужный коммит
git checkout 2db8540r7y13f09f36232e0u94c56e655bd79d79
//Переключиться на последний коммит в ветке master
git checkout master


//8. Пошаговая отмена изменений
//отмена незапушенного коммита
git reset --soft HEAD^
//отмена файла из индекса
git reset HEAD file.txt
//возврат до коммита
git checkout -- file.txt
(git reset --hard HEAD^ делает эти 3 шага за раз)
